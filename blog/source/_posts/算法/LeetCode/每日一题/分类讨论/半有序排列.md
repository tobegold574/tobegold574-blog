class Solution {
    public int semiOrderedPermutation(int[] nums) {
        int n=nums.length;
        int fir=0,last=0;
        int res;
        for(int i=0;i!=n;++i){
            if(nums[i]==1) fir=i;
            if(nums[i]==n) last=i;
        }
        // å°±æ˜¯åˆ†ç±»è®¨è®º
        if(fir>last) res=fir+n-1-last-1;
        else res=fir+n-1-last;
        return res;
        
    }
}
